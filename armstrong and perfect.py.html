armstrong and perfect.py 
def per():
    c=0
    n=int(input('Enter the number: '))
    p=n
    for i in range(1,n):
        if i%n==0:
            c=c+i
    if c==p:
        print('Its a perfect number')

def arm():
    n=int(input('Enter the number: '))
    s=n
    c=0
    while(n>0):
        k=n%10
        c=c+k**3
        n=n//10
    if s==c:
        print('Its an Armstrong number')

while True:
    print('1, perfect number')
    print('2, armstrong number')
    print('3, exit')
    ch=int(input('Enter option: '))
    if ch==1:
        per()
    elif ch==2:
        arm()
    else:
        break
